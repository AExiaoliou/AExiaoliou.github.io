ORIGIN PROBLEM

题目大意

题目意思其实很简单:
你有 nnn 个箱子按序号排在一条线上，就像这样

//assume n = 7
1 2 3 4 5 6 7

然后有4种操作, 用 1 2 3 4 表示操作类型
除了4之外后面都跟 xxx 和 yyy 表示操作的两个数
最后让你输出所有奇数位置的数的总和的大小
题目示例如下

//assume n = 6
1 2 3 4 5 6
//move x to the left to y
// 1 1 4
2 3 1 4 5 6
//move x to the right to y
// 2 3 5
2 1 4 5 3 6
//swap x and y
// 3 1 6
2 6 4 5 3 1
//reverse
// 4
1 3 5 4 6 2
sum = 1 + 5 + 6 = 12

看到这里如果想到用传统链表就错了，因为传统链表找到一个元素的时间是线性的，就算用迭代器去节约常数时间，但出题人只要不傻，很轻松就能出一组attacker数据
也就逆转链表有可能o(1)解决（但实际上大部分语言的内置实现都是o(n)的)
这里就要注意到题目的一个细节，这个数组一个是定长的，也就是说我们无需去关注容器的增减，只需要关注容器内元素之间的顺序就好了，也就是说 -> 模拟链表
我们可以开一个数组

new int[] {1, 2, 3, 4, 5, 6}

现在我们就可以o(1)访问了，但是我们怎么获得他们的顺序呢
可以这样

  = left       = left
1            2           3
    right =      right =

除此之外还有一个技巧保证o(1)逆转链表，看下面代码就懂，就不多加阐述了
接下来的事情就非常简单了，如果不会链表的基本操作的话，出门右转数据结构与算法分析

我的代码：

#include <cstdio>
#include <iostream>

using namespace std;

const int maxn = 100010;
bool inv = true;

struct node {
    node() = default;
    node(int a, int b) : _left(a), _right(b) {}
    int _left = 0; //这里做约定性隐藏就好，当时想着说不定可以用到
    int _right = 0;
    int & left() { return inv ? _left : _right; } //只需要在访问时判断inv就可以(o1)逆转链表
    int & right() { return inv ? _right : _left; }
} a[maxn];

void init(int n) {
    inv = true;
    a[0] = { n, 1 }; //访问第一个元素时，如果链表被逆转，访问的是第n个元素
    for (int i = 1; i < n; i++) {
        a[i] = { i - 1, i + 1 };
    }
    a[n] = { n - 1, 0 }; // 把0当做null，不过这里的另外一种意义是链接到链表首（好像不小心循环链表了
}
//下面的东西看起来很长，其实没啥技术含量，只要熟悉链表，手速快点，很快就能写完
//其实我这里也写得不太好，其实多弄几个temp就非常清晰了，这样就是省了点常数时间而已，没卵用
//(的确挺丑的)
void solve(int m) {
    for (int i = 0; i < m; i++) {
        int cmd, x, y;
        cin >> cmd;
        if (cmd == 1) {
            cin >> x >> y;
            if (a[x].right() == y) continue;
            //语法看起来有点恶心，但是可以用c的预处理器简化，有兴趣可以玩玩
            //（颅内自动脑补 x.left.right 也不是不行 （逃
            a[a[x].left()].right() = a[x].right();
            a[a[x].right()].left() = a[x].left();
            a[a[y].left()].right() = x;
            a[x].left() = a[y].left();
            a[x].right() = y;
            a[y].left() = x;
        }
        else if (cmd == 2) {
            cin >> x >> y;
            if (a[x].left() == y) continue;
            a[a[x].left()].right() = a[x].right();
            a[a[x].right()].left() = a[x].left();
            a[a[y].right()].left() = x;
            a[x].right() = a[y].right();
            a[x].left() = y;
            a[y].right() = x;
        }
        else if (cmd == 3) {
            cin >> x >> y;
            //这里有 x y 相邻的特殊情况，分别特判x在y左边，y在x左边写就好了
            if (a[x].right() == y) {
                int temp_left = a[x].left(), temp_right = a[y].right();
                a[temp_left].right() = y; a[temp_right].left() = x;
                a[x].left() = y; a[x].right() = temp_right;
                a[y].left() = temp_left; a[y].right() = x;
                continue;
            }
            else if (a[y].right() == x) {
                int temp_left = a[y].left(), temp_right = a[x].right();
                a[temp_left].right() = x; a[temp_right].left() = y;
                a[y].left() = x; a[y].right() = temp_right;
                a[x].left() = temp_left; a[x].right() = y;
                continue;
            }
            //正常写就好
            a[a[x].left()].right() = y;
            a[a[x].right()].left() = y;
            a[a[y].left()].right() = x;
            a[a[y].right()].left() = x;
            int temp_left_x = a[x].left(), temp_right_x = a[x].right();
            int temp_left_y = a[y].left(), temp_right_y = a[y].right();
            a[x].left() = temp_left_y;
            a[x].right() = temp_right_y;
            a[y].left() = temp_left_x;
            a[y].right() = temp_right_x;
        }
        else {
            inv = !inv;
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0);
    int n, m;
    int cases = 1;
    while (cin >> n >> m) {
        init(n);
        solve(m);
        int s = a[0].right();
        long long ans = s;
        for (int i = 3; i <= n; i += 2) {
            s = a[a[s].right()].right();
            ans += s;
        }
        cout << "Case " << cases << ": " << ans << endl;
        cases += 1;
    }
    return 0;
}
